条件判断

条件判断
  1. if
    void main() {
 bool flag = true;
 if ( flag ) {
   print('条件为真');
 }
}

  2. if双路分支

    void main() {
 bool flag = true;
 if ( flag ){
   print('🍇');
 } else {
   print('🍈');
 }
}

  3. if多路分支

    void main() {
 String type = 'A';
 if ( type === 'A' ){
   print('😂');
 } else if ( type === 'B') {
   print('😊');
 } else {
   print('😍');
 }
}
  4. swtich语句
    void main() {
 String str = "笑尿了";
 switch( str ){
   case '笑尿了':
     print('😂');
     break;
   case '笑里藏刀':
     print('😊');
     break;
   default: 
     print('😍');
     break;
 }
}
