运算符

运算符

一、算术运算符

算术运算符主要有： + - * / % ~/

  1. 加法
     mian() {
     int a = 3; 
     int b = 4;
     int c = a + b;
     print( c );
 }
  2. 减法
     mian() {
     int a = 3; 
     int b = 4;
     int c = a - b;
     print( c );
 }
  3. 乘法
     mian() {
     int a = 3; 
     int b = 4;
     int c = a * b;
     print( c );
 }
  4. 除法
     mian() {
     int a = 5; 
     int b = 2;
     int c = a / b;
     print( c ); // 2.5
 }

  5. 取余

     mian() {
     int a = 10; 
     int b = 3;
     int c = a % b;
     print( c ); // 3
 }

  6. 整除

     mian() {
     int a = 10; 
     int b = 3;
     int c = a ~/ b;
     print( c ); // 3
 }

二、关系运算符

关系运算符主要有： == 等于 != 不相等 > 大于 < 小于 >= 大于等于 <= 小于等于

  1. ==

     main() {
     int a = 10;
     int b = 20;
     if ( a == b ) {
         print('a等于b');
     }
     else {
         print('a不等于b');

     }
 }

  2. !=

     main() {
     int a = 10;
     int b = 20;
     if ( a != b ) {
         print('a不等于b');
     }
     else {
         print('a等于b');

     }
 }

  3. >

     main() {
     int a = 10;
     int b = 20;
     if ( a > b ) {
         print('a大于b');
     }
     else {
         print('a不大于b');

     }
 }
  4. <
     main() {
     int a = 10;
     int b = 20;
     if ( a < b ) {
         print('a小于b');
     }
     else {
         print('a不小于b');
     }
 }
  5. 大于等于：>=
     main() {
     int a = 30;
     int b = 20;
     if ( a >= b ) {
         print('a大于等于b');
     }
     else {
         print('a小于b');
     }
 }
  6. 小于等于 >=
     main() {
     int a = 10;
     int b = 20;
     if ( a <= b ) {
         print('a小于等于b');
     }
     else {
         print('a大于b');
     }
 }

三、逻辑运算符

逻辑运算符有： ! || &&

  1. 非 ！
     main() {
     bool flag = true;
     if ( !flag ) {
         print('flag为false');
     }
     else {
         print('flag为true');
     }
 }
  2. 与 ！
     main() {
     int a = 10;
     int b = 20;
     bool flag = true;
     if ( a > b && flag  ) {
         print('a大于b并且flag为true');
     }
     else {
         print('a小于等于b或是flag为false');
     }
 }
  3. 或 ||
     main() {
     int a = 10;
     int b = 20;
     bool flag = true;
     if ( a > b || flag  ) {
         print('a > b 成立或是flag为true');
     }
     else {
         print('a>b不成立或是flag为false');
     }
 }

四、赋值运算符

赋值运算符有：= += -= /= %= *= ??=

  1. = 
     main() {
     int a = 10;
     int b = 20;
     int c = a + b;
     print( c );
 }
  2. += 
     main() {
     int a = 10;
     a += 20;   //  a = a + 20 
     print( a );
 }
  3. -= 
     main() {
     int a = 10;
     a -= 20;  // a = a - 20 
     print( a );
 }
  4. *= 
     main() {
     int a = 10;
     a *= 20;  //   a = a * 20 
     print( a );
 }
  5. /= 
     main() {
     int a = 10;
     a /= 20;  //   a = a / 20 
     print( a );
 }
  6. %= 
     main() {
     int a = 10;
     a %= 20;  //   a = a % 20 
     print( a );
 }
  7. ??=
     main() {
     // Assign value to a
     a = value;
     // Assign value to b if b is null; otherwise, b stays the same
     // b为null时才会赋值，不为null不会改变b的值
     b ??= value;
     print( b );
 }
